def merge_intervals(intervals):
    if not intervals:
        return []
    intervals = sorted(intervals, key=lambda x: x[0])
    merged = [intervals[0][:]]
    for start, end in intervals[1:]:
        last = merged[-1]
        if start <= last[1]:          # overlap or adjacency
            last[1] = max(last[1], end)
        else:
            merged.append([start, end])
    return merged
