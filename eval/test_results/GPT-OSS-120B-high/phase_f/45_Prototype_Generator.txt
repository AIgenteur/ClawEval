from dataclasses import dataclass, asdict
from flask import Flask, request, jsonify, abort

app = Flask(__name__)

# In‑memory "database"
books_db = []
next_id = 1


@dataclass
class Book:
    id: int
    title: str
    author: str
    isbn: str
    price: float


def find_book(book_id: int):
    return next((b for b in books_db if b.id == book_id), None)


def validate_book_payload(payload, partial=False):
    required = {"title", "author", "isbn", "price"}
    if not isinstance(payload, dict):
        abort(400, description="Invalid JSON payload")
    missing = required - payload.keys() if not partial else set()
    if missing:
        abort(422, description=f"Missing fields: {', '.join(missing)}")
    for field in required:
        if field in payload:
            if field == "price":
                try:
                    float(payload[field])
                except (TypeError, ValueError):
                    abort(422, description="Price must be a number")
            elif not isinstance(payload[field], str):
                abort(422, description=f"{field} must be a string")


@app.errorhandler(400)
@app.errorhandler(404)
@app.errorhandler(422)
def handle_error(err):
    response = jsonify(error=err.description if err.description else err.name)
    response.status_code = err.code if hasattr(err, "code") else 500
    return response


@app.route("/books", methods=["GET"])
def list_books():
    page = int(request.args.get("page", 1))
    per_page = int(request.args.get("per_page", 10))
    start = (page - 1) * per_page
    end = start + per_page
    items = [asdict(b) for b in books_db[start:end]]
    return jsonify(items), 200


@app.route("/books/<int:book_id>", methods=["GET"])
def get_book(book_id):
    book = find_book(book_id)
    if not book:
        abort(404, description="Book not found")
    return jsonify(asdict(book)), 200


@app.route("/books", methods=["POST"])
def create_book():
    global next_id
    payload = request.get_json()
    validate_book_payload(payload)
    book = Book(
        id=next_id,
        title=payload["title"],
        author=payload["author"],
        isbn=payload["isbn"],
        price=float(payload["price"]),
    )
    books_db.append(book)
    next_id += 1
    return jsonify(asdict(book)), 201


@app.route("/books/<int:book_id>", methods=["PUT"])
def update_book(book_id):
    book = find_book(book_id)
    if not book:
        abort(404, description="Book not found")
    payload = request.get_json()
    validate_book_payload(payload, partial=False)
    book.title = payload["title"]
    book.author = payload["author"]
    book.isbn = payload["isbn"]
    book.price = float(payload["price"])
    return jsonify(asdict(book)), 200


@app.route("/books/<int:book_id>", methods=["DELETE"])
def delete_book(book_id):
    book = find_book(book_id)
    if not book:
        abort(404, description="Book not found")
    books_db.remove(book)
    return "", 204


if __name__ == "__main__":
    app.run(debug=True)
