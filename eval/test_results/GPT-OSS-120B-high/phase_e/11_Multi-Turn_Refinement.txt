import re

def parse_duration(s: str) -> int:
    if not s:
        return -1
    pattern = re.compile(r'(\d+)([hms])')
    total = 0
    pos = 0
    for match in pattern.finditer(s):
        if match.start() != pos:   # gap or unexpected character
            return -1
        value = int(match.group(1))
        unit = match.group(2)
        if unit == 'h':
            total += value * 3600
        elif unit == 'm':
            total += value * 60
        elif unit == 's':
            total += value
        pos = match.end()
    if pos != len(s):               # leftover characters
        return -1
    return total

```python
import re

def parse_duration(s: str) -> int:
    if not s:
        return -1
    pattern = re.compile(r'(\d+)([hms])')
    total = 0
    pos = 0
    seen = set()
    for match in pattern.finditer(s):
        if match.start() != pos:   # gap or unexpected character
            return -1
        value = int(match.group(1))
        unit = match.group(2)
        if unit in seen:            # duplicate unit
            return -1
        seen.add(unit)
        if unit == 'h':
            total += value * 3600
        elif unit == 'm':
            total += value * 60
        elif unit == 's':
            total += value
        pos = match.end()
    if pos != len(s):               # leftover characters
        return -1
    return total
```
