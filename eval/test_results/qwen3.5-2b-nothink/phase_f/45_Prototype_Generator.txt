```python
from flask import Flask, request, jsonify
from datetime import datetime
from typing import Optional, Dict, Any

app = Flask(__name__)

# In-memory database
books_db: list[Dict[str, Any]] = []

class Book:
    def __init__(self, title: str, author: str, isbn: str, price: float):
        self.id = None
        self.title = title
        self.author = author
        self.isbn = isbn
        self.price = price
        self.created_at = datetime.now().isoformat()

    def to_dict(self) -> Dict[str, Any]:
        return {
            "id": self.id,
            "title": self.title,
            "author": self.author,
            "isbn": self.isbn,
            "price": self.price,
            "created_at": self.created_at
        }

def validate_book_data(data: Dict[str, Any]) -> Dict[str, Any]:
    """Input validation helper"""
    required_fields = ['title', 'author', 'isbn', 'price']
    missing_fields = [field for field in required_fields if field not in data]
    
    if missing_fields:
        return {"error": f"Missing required fields: {', '.join(missing_fields)}"}
    
    # Basic type checking for price
    if not isinstance(data['price'], (int, float)) or data['price'] <= 0:
        return {"error": "Price must be a positive number"}
    
    return {}

@app.route('/books', methods=['GET'])
def get_books():
    page = request.args.get('page', 1, type=int)
    per_page = request.args.get('per_page', 10, type=int)
    
    if page < 1:
        page = 1
    if per_page < 1:
        per_page = 10
    
    start = (page - 1) * per_page
    end = start + per_page
    
    books = books_db[start:end]
    
    # Pagination metadata
    total = len(books_db)
    total_pages = (total + per_page - 1) // per_page if total > 0 else 0
    
    return jsonify({
        "books": [b.to_dict() for b in books],
        "pagination": {
            "page": page,
            "per_page": per_page,
            "total": total,
            "total_pages": total_pages
        }
    })

@app.route('/books/<int:book_id>', methods=['GET'])
def get_book(book_id: int):
    book = next((b for b in books_db if b.id == book_id), None)
    
    if book is None:
        return jsonify({"error": "Book not found"}), 404
    
    return jsonify(book.to_dict())

@app.route('/books', methods=['POST'])
def create_book():
    data = request.get_json()
    
    if not data:
        return jsonify({"error": "Request body must be JSON"}), 400
    
    validation_errors = validate_book_data(data)
    if validation_errors:
        return jsonify(validation_errors), 422
    
    new_book = Book(
        title=data['title'],
        author=data['author'],
        isbn=data['isbn'],
        price=data['price']
    )
    
    books_db.append(new_book)
    
    return jsonify(new_book.to_dict()), 201

@app.route('/books/<int:book_id>', methods=['PUT'])
def update_book(book_id: int):
    book = next((b for b in books_db if b.id == book_id), None)
    
    if book is None:
        return jsonify({"error": "Book not found"}), 404
    
    data = request.get_json()
    
    if not data:
        return jsonify({"error": "Request body must be JSON"}), 400
    
    validation_errors = validate_book_data(data)
    if validation_errors:
        return jsonify(validation_errors), 422
    
    # Update fields
    if 'title' in data:
        book.title = data['title']
    if 'author' in data:
        book.author = data['author']
    if 'isbn' in data:
        book.isbn = data['isbn']
    if 'price' in data:
        book.price = data['price']
    
    return jsonify(book.to_dict()), 200

@app.route('/books/<int:book_id>', methods=['DELETE'])
def delete_book(book_id: int):
    book = next((b for b in books_db if b.id == book_id), None)
    
    if book is None:
        return jsonify({"error": "Book not found"}), 404
    
    books_db.remove(book)
    
    return jsonify({"message": "Book deleted"}), 200

if __name__ == '__main__':
    app.run(debug=True)
```
