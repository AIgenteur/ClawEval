def merge_intervals(intervals):
    if not intervals:
        return []

    # Sort by start time
    intervals.sort(key=lambda x: x[0])

    merged = [intervals[0]]

    for i in range(1, len(intervals)):
        last = merged[-1]
        curr = intervals[i]

        if curr[0] <= last[1]:  # Overlapping or adjacent
            last[1] = max(last[1], curr[1])
        else:
            merged.append(curr)

    return merged
