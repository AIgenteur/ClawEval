from collections import OrderedDict

def lru_cache(capacity):
    """
    Returns an LRU cache with the specified capacity.
    Operations are O(1).
    """
    cache = OrderedDict()
    
    def get(key):
        if key in cache:
            cache.move_to_end(key)  # Move to end = most recently used
        else:
            cache[key] = -1  # Mark as not found
        return cache[key]
    
    def put(key, value):
        if key in cache:
            cache.move_to_end(key)  # Update position
        else:
            if len(cache) >= capacity:
                # Evict least recently used (first item)
                cache.popitem(last=False)
            cache[key] = value
            cache.move_to_end(key)  # Mark as most recently used
    
    return cache
