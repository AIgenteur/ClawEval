def test_calculate_tax_single_brackets():
    """Test calculation for single filer using brackets_single."""
    result = calculate_tax(100000, 'single')
    assert result == (100000 * 0.10, 0.10)
    result = calculate_tax(200000, 'single')
    assert result == (200000 * 0.12, 0.12)
    result = calculate_tax(300000, 'single')
    assert result == (300000 * 0.22, 0.22)
    result = calculate_tax(400000, 'single')
    assert result == (400000 * 0.24, 0.24)
    result = calculate_tax(500000, 'single')
    assert result == (500000 * 0.32, 0.32)
    result = calculate_tax(600000, 'single')
    assert result == (600000 * 0.35, 0.35)
    result = calculate_tax(700000, 'single')
    assert result == (700000 * 0.37, 0.37)


def test_calculate_tax_married_brackets():

def test_calculate_tax_bracket_boundaries():
    """Test calculation at exact bracket boundaries."""
    # Single: 11600 -> 0.10, 47150 -> 0.12
    result = calculate_tax(11600, 'single')
    assert result == (0.10, 0.10)
    result = calculate_tax(47150, 'single')
    assert result == (0.12, 0.12)
    
    # Single: 100525 -> 0.22, 191950 -> 0.24
    result = calculate_tax(100525, 'single')
    assert result == (0.22, 0.22)
    result = calculate_tax(191950, 'single')
    assert result == (0.24, 0.24)
    
    # Single: 243725 -> 0.32, 609350 -> 0.35
    result = calculate_tax(243725, 'single')
    assert result == (0.32, 0.32)
    result = calculate_tax(609350, 'single')
    assert result == (0.35, 0.35)
    
    # Single: float('inf') -> 0.37
    result = calculate_tax(float('inf'), 'single')
    assert result == (0.37, 0.37)
