from typing import Any, Callable, Dict, List, Tuple

def lru_cache(capacity: int) -> Callable[[str, Any], Any]:
    """
    Returns a cache object with get(key) and put(key, value) methods.
    
    - get(key): returns value if exists, -1 if not
    - put(key, value): inserts/updates. If at capacity, evicts least recently used.
    - Both get and put count as "use" for LRU purposes.
    """
    cache: Dict[str, Any] = {}
    
    def get(key: str) -> Any:
        if key in cache:
            return cache[key]
        else:
            return -1
