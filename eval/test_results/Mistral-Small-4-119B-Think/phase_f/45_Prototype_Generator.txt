```python
from flask import Flask, request, jsonify
from dataclasses import dataclass
from typing import List, Optional, Dict, Any
import uuid

app = Flask(__name__)

@dataclass
class Book:
    id: str
    title: str
    author: str
    isbn: str
    price: float

books_db: List[Book] = []

def validate_book_data(data: Dict[str, Any]) -> Optional[str]:
    if not data.get('title'):
        return "Title is required"
    if not data.get('author'):
        return "Author is required"
    if not data.get('isbn'):
        return "ISBN is required"
    if not data.get('price') or data.get('price') <= 0:
        return "Price must be a positive number"
    return None

@app.route('/books', methods=['GET'])
def get_books():
    page = int(request.args.get('page', 1))
    per_page = int(request.args.get('per_page', 10))

    start = (page - 1) * per_page
    end = start + per_page
    paginated_books = books_db[start:end]

    return jsonify({
        'books': [vars(book) for book in paginated_books],
        'page': page,
        'per_page': per_page,
        'total': len(books_db)
    })

@app.route('/books/<book_id>', methods=['GET'])
def get_book(book_id):
    book = next((b for b in books_db if b.id == book_id), None)
    if not book:
        return jsonify({'error': 'Book not found'}), 404
    return jsonify(vars(book))

@app.route('/books', methods=['POST'])
def create_book():
    data = request.get_json()
    error = validate_book_data(data)
    if error:
        return jsonify({'error': error}), 400

    book = Book(
        id=str(uuid.uuid4()),
        title=data['title'],
        author=data['author'],
        isbn=data['isbn'],
        price=float(data['price'])
    )
    books_db.append(book)
    return jsonify(vars(book)), 201

@app.route('/books/<book_id>', methods=['PUT'])
def update_book(book_id):
    book = next((b for b in books_db if b.id == book_id), None)
    if not book:
        return jsonify({'error': 'Book not found'}), 404

    data = request.get_json()
    error = validate_book_data(data)
    if error:
        return jsonify({'error': error}), 400

    book.title = data['title']
    book.author = data['author']
    book.isbn = data['isbn']
    book.price = float(data['price'])
    return jsonify(vars(book))

@app.route('/books/<book_id>', methods=['DELETE'])
def delete_book(book_id):
    global books_db
    book = next((b for b in books_db if b.id == book_id), None)
    if not book:
        return jsonify({'error': 'Book not found'}), 404

    books_db = [b for b in books_db if b.id != book_id]
    return jsonify({'message': 'Book deleted'}), 200

if __name__ == '__main__':
    app.run(debug=True)
```
