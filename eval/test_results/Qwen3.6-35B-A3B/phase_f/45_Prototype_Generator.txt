```python
from flask import Flask, request, jsonify
from dataclasses import dataclass, asdict
from typing import List
import uuid

app = Flask(__name__)

@dataclass
class Book:
    id: str
    title: str
    author: str
    isbn: str
    price: float

books_db: List[Book] = []

def validate_book_data(data):
    required_fields = ['title', 'author', 'isbn', 'price']
    missing = [f for f in required_fields if f not in data or data[f] is None]
    if missing:
        return None, f"Missing required fields: {', '.join(missing)}"

    if not isinstance(data['title'], str) or not data['title'].strip():
        return None, "Title must be a non-empty string"
    if not isinstance(data['author'], str) or not data['author'].strip():
        return None, "Author must be a non-empty string"
    if not isinstance(data['isbn'], str) or not data['isbn'].strip():
        return None, "ISBN must be a non-empty string"

    try:
        price = float(data['price'])
        if price < 0:
            return None, "Price must be a non-negative number"
    except (ValueError, TypeError):
        return None, "Price must be a valid number"

    return {
        'title': data['title'].strip(),
        'author': data['author'].strip(),
        'isbn': data['isbn'].strip(),
        'price': price
    }, None

@app.route('/books', methods=['GET'])
def get_books():
    page = request.args.get('page', 1, type=int) or 1
    per_page = request.args.get('per_page', 10, type=int) or 10
    page = max(1, page)
    per_page = max(1, min(100, per_page))

    start = (page - 1) * per_page
    end = start + per_page
    paginated_books = books_db[start:end]

    return jsonify([asdict(b) for b in paginated_books]), 200

@app.route('/books/<string:book_id>', methods=['GET'])
def get_book(book_id):
    book = next((b for b in books_db if b.id == book_id), None)
    if not book:
        return jsonify({"error": "Book not found"}), 404
    return jsonify(asdict(book)), 200

@app.route('/books', methods=['POST'])
def create_book():
    data = request.get_json(silent=True)
    if not data:
        return jsonify({"error": "Request body must be valid JSON"}), 400

    book_data, error = validate_book_data(data)
    if error:
        return jsonify({"error": error}), 422

    new_book = Book(id=str(uuid.uuid4()), **book_data)
    books_db.append(new_book)
    return jsonify(asdict(new_book)), 201

@app.route('/books/<string:book_id>', methods=['PUT'])
def update_book(book_id):
    book = next((b for b in books_db if b.id == book_id), None)
    if not book:
        return jsonify({"error": "Book not found"}), 404

    data = request.get_json(silent=True)
    if not data:
        return jsonify({"error": "Request body must be valid JSON"}), 400

    book_data, error = validate_book_data(data)
    if error:
        return jsonify({"error": error}), 422

    book.title = book_data['title']
    book.author = book_data['author']
    book.isbn = book_data['isbn']
    book.price = book_data['price']

    return jsonify(asdict(book)), 200

@app.route('/books/<string:book_id>', methods=['DELETE'])
def delete_book(book_id):
    global books_db
    book = next((b for b in books_db if b.id == book_id), None)
    if not book:
        return jsonify({"error": "Book not found"}), 404

    books_db = [b for b in books_db if b.id != book_id]
    return jsonify({"message": "Book deleted"}), 200

if __name__ == '__main__':
    app.run(debug=True)
```
