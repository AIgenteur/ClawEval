class LRUCache:
    class Node:
        def __init__(self, key, value):
            self.key = key
            self.value = value
            self.prev = None
            self.next = None

    def __init__(self, capacity):
        self.capacity = capacity
        self.cache = {}
        self.head = self.Node(0, 0)
        self.tail = self.Node(0, 0)
        self.head.next = self.tail
        self.tail.prev = self.head

    def _add_to_head(self, node):
        node.prev = self.head
        node.next = self.head.next
        self.head.next.prev = node
        self.head.next = node

    def _remove_node(self, node):
        node.prev.next = node.next
        node.next.prev = node.prev

    def _move_to_head(self, node):
        self._remove_node(node)
        self._add_to_head(node)

    def _pop_tail(self):
        res = self.tail.prev
        self._remove_node(res)
        return res

    def get(self, key):
        if key in self.cache:
            node = self.cache[key]
            self._move_to_head(node)
            return node.value
        return -1

    def put(self, key, value):
        if key in self.cache:
            node = self.cache[key]
            node.value = value
            self._move_to_head(node)
        else:
            node = self.Node(key, value)
            self.cache[key] = node
            self._add_to_head(node)
            if len(self.cache) > self.capacity:
                tail_node = self._pop_tail()
                del self.cache[tail_node.key]


class MinStack:
    def __init__(self):
        self.stack = []

    def push(self, val):
        if not self.stack:
            self.stack.append((val, val))
        else:
            current_min = val if val < self.stack[-1][1] else self.stack[-1][1]
            self.stack.append((val, current_min))

    def pop(self):
        val, _ = self.stack.pop()
        return val

    def top(self):
        return self.stack[-1][0]

    def get_min(self):
        return self.stack[-1][1]


def find_median(stream):
    sorted_nums = []
    medians = []
    for num in stream:
        # Binary search for insertion point
        lo, hi = 0, len(sorted_nums)
        while lo < hi:
            mid = (lo + hi) // 2
            if sorted_nums[mid] < num:
                lo = mid + 1
            else:
                hi = mid
        sorted_nums.insert(lo, num)
        
        n = len(sorted_nums)
        if n % 2 == 1:
            medians.append(float(sorted_nums[n // 2]))
        else:
            medians.append((sorted_nums[n // 2 - 1] + sorted_nums[n // 2]) / 2.0)
    return medians
