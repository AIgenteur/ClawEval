def is_valid_password(password):
    if not (8 <= len(password) <= 64):
        return False
    if not any(c.isupper() for c in password):
        return False
    if not any(c.islower() for c in password):
        return False
    if not any(c.isdigit() for c in password):
        return False
    special_chars = "!@#$%^&*()-_+="
    if not any(c in special_chars for c in password):
        return False
    if ' ' in password:
        return False
    if password[0].isdigit():
        return False
    if any(password[i] == password[i+1] == password[i+2] == password[i+3] for i in range(len(password)-3)):
        return False
    if "password" in password.lower():
        return False
    return True

def run_tests():
    tests = [
        ("Str0ng!Pass", True),
        ("Ab1!cde", False),
        ("A" * 65 + "1!a", False),
        ("abc123!a", False),
        ("ABC123!A", False),
        ("Abcdef!@", False),
        ("Abcdef12", False),
        ("Abcdef1! ", False),
        ("1Abcdef!@", False),
        ("Abcdefg1!!!!", False),
        ("Abcdefg1!Password", False),
        ("Abcdefg1!p@ssword", False),
        ("A1!aBcDe", True),
        ("A1!aBcDe" + "x" * 56, True),
        ("Abcdefg1!!!", True)
    ]
    return [(inp, exp, is_valid_password(inp)) for inp, exp in tests]
