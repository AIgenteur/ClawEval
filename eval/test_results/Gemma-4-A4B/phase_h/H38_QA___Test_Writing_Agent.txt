def is_valid_password(password):
    # Rule: At least 8 characters long and at most 64 characters
    if not (8 <= len(password) <= 64):
        return False
    
    # Rule: Contains at least 1 uppercase letter
    if not any(c.isupper() for c in password):
        return False
    
    # Rule: Contains at least 1 lowercase letter
    if not any(c.islower() for c in password):
        return False
    
    # Rule: Contains at least 1 digit
    if not any(c.isdigit() for c in password):
        return False
    
    # Rule: Contains at least 1 special character from: !@#$%^&*()-_+=
    special_chars = "!@#$%^&*()-_+="
    if not any(c in special_chars for c in password):
        return False
    
    # Rule: No spaces allowed
    if ' ' in password:
        return False
    
    # Rule: Cannot start with a digit
    if password[0].isdigit():
        return False
    
    # Rule: Cannot contain the word "password" (case-insensitive)
    if "password" in password.lower():
        return False
    
    # Rule: Cannot contain more than 3 consecutive identical characters
    consecutive_count = 1
    for i in range(1, len(password)):
        if password[i] == password[i-1]:
            consecutive_count += 1
            if consecutive_count > 3:
                return False
        else:
            consecutive_count = 1
            
    return True

def run_tests():
    test_cases = [
        ("Valid1!abc", True),               # Valid case
        ("Short1!", False),                 # Too short
        ("A" * 65 + "a1!", False),          # Too long
        ("lowercase1!", False),             # No uppercase
        ("UPPERCASE1!", False),             # No lowercase
        ("NoDigit!abc", False),             # No digit
        ("NoSpecial1abc", False),           # No special character
        ("With Space1!", False),            # Contains space
        ("1StartWithDigit!a", False),       # Starts with digit
        ("Abbbbc1!", False),                # 4 consecutive identical characters
        ("Password123!", False),            # Contains "password" (mixed case)
        ("pAsSwOrD123!", False),            # Contains "password" (mixed case)
        ("Abbb1!abc", True),                # 3 consecutive identical characters (allowed)
        ("!@#$%^&*()_+", False),            # Missing upper, lower, and digit
        ("A1!bcdefghijk", True)             # Valid case
    ]
    
    results = []
    for inp, expected in test_cases:
        actual = is_valid_password(inp)
        results.append((inp, expected, actual))
    return results
