```python
from flask import Flask, request, jsonify
from dataclasses import dataclass, asdict
from typing import List, Optional
import uuid

app = Flask(__name__)

@dataclass
class Book:
    id: str
    title: str
    author: str
    isbn: str
    price: float

# In-memory database
books_db: List[Book] = []

def validate_book_data(data: dict, partial: bool = False) -> Optional[str]:
    """Validates book data. If partial is True, it allows missing fields (for PUT)."""
    required_fields = {
        "title": str,
        "author": str,
        "isbn": str,
        "price": (int, float)
    }
    
    if not partial:
        for field in required_fields:
            if field not in data:
                return f"Missing required field: {field}"
    
    for field, expected_type in required_fields.items():
        if field in data:
            if not isinstance(data[field], expected_type):
                return f"Invalid type for {field}. Expected {expected_type}"
    return None

@app.route('/books', methods=['GET'])
def get_books():
    try:
        page = int(request.args.get('page', 1))
        per_page = int(request.args.get('per_page', 10))
    except ValueError:
        return jsonify({"error": "page and per_page must be integers"}), 400

    if page < 1 or per_page < 1:
        return jsonify({"error": "page and per_page must be positive integers"}), 400

    start = (page - 1) * per_page
    end = start + per_page
    
    paginated_books = [asdict(b) for b in books_db[start:end]]
    
    return jsonify({
        "total": len(books_db),
        "page": page,
        "per_page": per_page,
        "books": paginated_books
    }), 200

@app.route('/books/<string:book_id>', methods=['GET'])
def get_book(book_id):
    book = next((b for b in books_db if b.id == book_id), None)
    if not book:
        return jsonify({"error": "Book not found"}), 404
    return jsonify(asdict(book)), 200

@app.route('/books', methods=['POST'])
def create_book():
    data = request.get_json()
    if not data:
        return jsonify({"error": "Invalid JSON"}), 400
    
    error_msg = validate_book_data(data)
    if error_msg:
        return jsonify({"error": error_msg}), 422

    new_book = Book(
        id=str(uuid.uuid4()),
        title=data['title'],
        author=data['author'],
        isbn=data['isbn'],
        price=float(data['price'])
    )
    books_db.append(new_book)
    return jsonify(asdict(new_book)), 201

@app.route('/books/<string:book_id>', methods=['PUT'])
def update_book(book_id):
    book = next((b for b in books_db if b.id == book_id), None)
    if not book:
        return jsonify({"error": "Book not found"}), 404

    data = request.get_json()
    if not data:
        return jsonify({"error": "Invalid JSON"}), 400

    error_msg = validate_book_data(data, partial=True)
    if error_msg:
        return jsonify({"error": error_msg}), 422

    # Update fields
    book.title = data.get('title', book.title)
    book.author = data.get('author', book.author)
    book.isbn = data.get('isbn', book.isbn)
    book.price = float(data.get('price', book.price))

    return jsonify(asdict(book)), 200

@app.route('/books/<string:book_id>', methods=['DELETE'])
def delete_book(book_id):
    global books_db
    book = next((b for b in books_db if b.id == book_id), None)
    if not book:
        return jsonify({"error": "Book not found"}), 404

    books_db = [b for b in books_db if b.id != book_id]
    return jsonify({"message": "Book deleted successfully"}), 200

if __name__ == '__main__':
    app.run(debug=True)
```
