def merge_overlapping(intervals):
    if not intervals:
        return []
    sorted_intervals = sorted(intervals, key=lambda x: x[0])
    merged = [sorted_intervals[0][:]]
    for current in sorted_intervals[1:]:
        prev = merged[-1]
        if current[0] <= prev[1]:
            prev[1] = max(prev[1], current[1])
        else:
            merged.append(current[:])
    return merged

def eval_rpn(tokens):
    if not tokens:
        return 0
    stack = []
    for token in tokens:
        if token in ('+', '-', '*', '/'):
            b = stack.pop()
            a = stack.pop()
            if token == '+':
                stack.append(a + b)
            elif token == '-':
                stack.append(a - b)
            elif token == '*':
                stack.append(a * b)
            else:  # '/'
                stack.append(int(a / b))
        else:
            stack.append(int(token))
    return stack[0]

def deep_flatten(nested):
    result = []
    def _flatten(item):
        if isinstance(item, (list, tuple)):
            for sub in item:
                _flatten(sub)
        else:
            result.append(item)
    _flatten(nested)
    return result
