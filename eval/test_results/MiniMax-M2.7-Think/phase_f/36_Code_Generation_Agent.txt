def merge_intervals(intervals):
    if not intervals:
        return []

    # Sort intervals by start time
    sorted_ints = sorted(intervals, key=lambda x: x[0])

    merged = [sorted_ints[0][:]]  # shallow copy of the first interval

    for start, end in sorted_ints[1:]:
        # If current interval overlaps or is adjacent to the last merged interval
        if start <= merged[-1][1]:
            merged[-1][1] = max(merged[-1][1], end)
        else:
            merged.append([start, end])

    return merged
