```python
from flask import Flask, request, jsonify
from dataclasses import dataclass, asdict
from typing import List

app = Flask(__name__)

# In-memory "database"
books_db: List['Book'] = []
next_id = 1

@dataclass
class Book:
    id: int
    title: str
    author: str
    isbn: str
    price: float

    def to_dict(self):
        return asdict(self)

    @classmethod
    def from_dict(cls, data: dict) -> 'Book':
        """Create a Book from a dict; expects keys title, author, isbn, price."""
        try:
            price = float(data['price'])
        except (KeyError, TypeError, ValueError):
            raise ValueError('price must be a numeric value')
        return cls(
            id=0,  # placeholder; will be set by the caller
            title=data['title'],
            author=data['author'],
            isbn=data['isbn'],
            price=price
        )


# ----------------------------------------------------------------------
# Helper utilities
# ----------------------------------------------------------------------
def _error_response(message: str, status_code: int):
    return jsonify({'error': message}), status_code


def _validate_required_fields(data: dict, required: List[str]):
    missing = [f for f in required if f not in data or not data[f]]
    if missing:
        raise ValueError(f"Missing required fields: {', '.join(missing)}")


def _validate_book_fields(data: dict):
    """Validate fields present in a book creation payload."""
    _validate_required_fields(data, ['title', 'author', 'isbn', 'price'])
    if not isinstance(data['title'], str) or not data['title'].strip():
        raise ValueError('title must be a non‑empty string')
    if not isinstance(data['author'], str) or not data['author'].strip():
        raise ValueError('author must be a non‑empty string')
    if not isinstance(data['isbn'], str) or not data['isbn'].strip():
        raise ValueError('isbn must be a non‑empty string')
    try:
        float(data['price'])
    except (TypeError, ValueError):
        raise ValueError('price must be a numeric value')


# ----------------------------------------------------------------------
# Routes
# ----------------------------------------------------------------------
@app.route('/books', methods=['GET'])
def list_books():
    """List all books with pagination."""
    page = request.args.get('page', 1, type=int)
    per_page = request.args.get('per_page', 10, type=int)

    if page < 1:
        return _error_response('page must be >= 1', 400)
    if per_page < 1:
        return _error_response('per_page must be >= 1', 400)

    total = len(books_db)
    start = (page - 1) * per_page
    end = start + per_page
    books_slice = books_db[start:end]

    return jsonify({
        'books': [b.to_dict() for b in books_slice],
        'page': page,
        'per_page': per_page,
        'total': total
    }), 200


@app.route('/books/<int:book_id>', methods=['GET'])
def get_book(book_id: int):
    """Retrieve a single book by its ID."""
    book = next((b for b in books_db if b.id == book_id), None)
    if book is None:
        return _error_response('Book not found', 404)
    return jsonify(book.to_dict()), 200


@app.route('/books', methods=['POST'])
def create_book():
    """Create a new book."""
    data = request.get_json(silent=True)
    if not data:
        return _error_response('Request body must be JSON', 400)

    try:
        _validate_book_fields(data)
    except ValueError as e:
        # Return 422 for validation errors that are semantic (e.g., missing fields,
        # wrong types). 400 can also be used; 422 is more precise for unprocessable
        # entity due to validation failure.
        return _error_response(str(e), 422)

    global next_id
    new_book = Book(
        id=next_id,
        title=data['title'].strip(),
        author=data['author'].strip(),
        isbn=data['isbn'].strip(),
        price=float(data['price'])
    )
    next_id += 1
    books_db.append(new_book)

    return jsonify(new_book.to_dict()), 201


@app.route('/books/<int:book_id>', methods=['PUT'])
def update_book(book_id: int):
    """Update an existing book (partial updates allowed)."""
    book = next((b for b in books_db if b.id == book_id), None)
    if book is None:
        return _error_response('Book not found', 404)

    data = request.get_json(silent=True)
    if not data:
        return _error_response('Request body must be JSON', 400)

    allowed = {'title', 'author', 'isbn', 'price'}
    invalid = set(data.keys()) - allowed
    if invalid:
        return _error_response(f"Invalid field(s): {', '.join(invalid)}", 400)

    try:
        if 'title' in data:
            if not isinstance(data['title'], str) or not data['title'].strip():
                raise ValueError('title must be a non‑empty string')
            book.title = data['title'].strip()

        if 'author' in data:
            if not isinstance(data['author'], str) or not data['author'].strip():
                raise ValueError('author must be a non‑empty string')
            book.author = data['author'].strip()

        if 'isbn' in data:
            if not isinstance(data['isbn'], str) or not data['isbn'].strip():
                raise ValueError('isbn must be a non‑empty string')
            book.isbn = data['isbn'].strip()

        if 'price' in data:
            try:
                book.price = float(data['price'])
            except (TypeError, ValueError):
                raise ValueError('price must be a numeric value')
    except ValueError as e:
        return _error_response(str(e), 422)

    return jsonify(book.to_dict()), 200


@app.route('/books/<int:book_id>', methods=['DELETE'])
def delete_book(book_id: int):
    """Delete a book by its ID."""
    global books_db
    book = next((b for b in books_db if b.id == book_id), None)
    if book is None:
        return _error_response('Book not found', 404)

    books_db = [b for b in books_db if b.id != book_id]
    return jsonify({'message': 'Book deleted'}), 200


# ----------------------------------------------------------------------
# Run the app
# ----------------------------------------------------------------------
if __name__ == '__main__':
    app.run(debug=True)
```
