def is_valid_password(password):
    if not isinstance(password, str):
        return False
    if len(password) < 8 or len(password) > 64:
        return False
    if ' ' in password:
        return False
    if password[0].isdigit():
        return False
    if 'password' in password.lower():
        return False
    if any(password[i] == password[i+1] == password[i+2] == password[i+3] for i in range(len(password)-3)):
        return False
    has_upper = any(c.isupper() for c in password)
    has_lower = any(c.islower() for c in password)
    has_digit = any(c.isdigit() for c in password)
    special_chars = set('!@#$%^&*()-_+=')
    has_special = any(c in special_chars for c in password)
    return has_upper and has_lower and has_digit and has_special

def run_tests():
    test_cases = [
        # Valid passwords
        ("Aa1!bcde", True),  # 8 chars, meets all
        ("Abcdefg1!", True), # 9 chars
        ("MyPass123!", True), # 9 chars
        ("A1b!A1b!A1b!A1b!A1b!A1b!A1b!A1b!A1b!A1b!A1b!A1b!A1b!A1b!A1b!A1b!A1b!A1b!A1b!", True), # 64 chars, valid
        ("Aa1!aaaa", True),  # 8 chars, 4 'a's but not consecutive 4? Wait: 'aaaa' is 4 consecutive -> INVALID! Fix below.
        # Actually, "Aa1!aaaa" has 'aaaa' at end -> 4 consecutive 'a' -> INVALID. So we need a valid one with max 3 repeats.
        ("Aa1!aaa", False),  # 7 chars -> too short
        ("Aa1!aaab", True),  # 8 chars: 'aaa' is 3 consecutive -> allowed
        # Invalid: length
        ("Aa1!", False),     # 4 chars
        ("A" * 65, False),   # 65 chars
        # Invalid: space
        ("Aa1! bcde", False),
        # Invalid: starts with digit
        ("1Aa!bcde", False),
        # Invalid: contains "password" (case-insensitive)
        ("Password1!", False),
        ("pAsSwOrD1!", False),
        ("MYPASSWORD1!", False),
        # Invalid: missing uppercase
        ("aa1!bcde", False),
        # Invalid: missing lowercase
        ("AA1!BCDE", False),
        # Invalid: missing digit
        ("Aa!bcdefg", False),
        # Invalid: missing special
        ("Aa1bcdefg", False),
        # Invalid: 4 consecutive identical
        ("Aa1!aaaa", False), # 4 'a's
        ("A!!!!bcde", False), # 4 '!'s
        ("Aaaaa1!", False),   # 5 'a's -> has 4 consecutive
        # Edge: exactly 3 consecutive allowed
        ("Aa1!aaab", True),   # as above
        ("Aa1!bbba", True),   # 3 'b's
        # Edge: 64 chars valid
        ("A" + "a1!" * 21, True), # 1 + 21*4 = 85 -> too long. Let's make 64.
        # We'll construct a 64-char valid one: "Aa1!" repeated 16 times = 64 chars
        ("Aa1!" * 16, True),
        # 65 chars: invalid
        ("Aa1!" * 16 + "A", False),
    ]
    # Ensure we have exactly 15 test cases
    test_cases = test_cases[:15]
    results = []
    for inp, expected in test_cases:
        actual = is_valid_password(inp)
        results.append((inp, expected, actual))
    return results
