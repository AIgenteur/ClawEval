def is_valid_password(password):
    # 1. Length check
    if len(password) < 8 or len(password) > 64:
        return False

    # 2. No spaces
    if ' ' in password:
        return False

    # 3. Cannot start with a digit
    if password and password[0].isdigit():
        return False

    # 4. No more than 3 consecutive identical characters
    for i in range(len(password) - 3):
        if password[i] == password[i+1] == password[i+2] == password[i+3]:
            return False

    # 5. Cannot contain the word "password" (case‑insensitive)
    if "password" in password.lower():
        return False

    # 6. At least one uppercase letter
    if not any(c.isupper() for c in password):
        return False

    # 7. At least one lowercase letter
    if not any(c.islower() for c in password):
        return False

    # 8. At least one digit
    if not any(c.isdigit() for c in password):
        return False

    # 9. At least one special character from the allowed set
    specials = "!@#$%^&*()-_+="
    if not any(c in specials for c in password):
        return False

    return True


def run_tests():
    tests = [
