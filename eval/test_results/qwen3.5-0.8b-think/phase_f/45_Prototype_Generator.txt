from flask import Flask, request, jsonify
from datetime import datetime
from typing import List, Optional
import uuid

app = Flask(__name__)

# In-memory database
books: List[dict] = []

# Book dataclass
class Book:
    id: str
    title: str
    author: str
    isbn: str
    price: float
    created_at: str

# Helper function to create a new book
def create_book(book_data: dict) -> Book:
    book = Book(
        id=str(uuid.uuid4()),
        title=book_data.get('title', ''),
        author=book_data.get('author', ''),
        isbn=book_data.get('isbn', ''),
        price=book_data.get('price', 0),
        created_at=datetime.now().isoformat()
    )
    books.append(book)
    return book

# Helper function to delete a book
def delete_book(book_id: str) -> None:
    books = [b for b in books if b.id != book_id]
    if books:
        books.pop()
    else:
        raise ValueError("Book not found")

# Helper function to update a book
def update_book(book_id: str, book_data: dict) -> Book:
    book = None
    for b in books:
        if b.id == book_id:
            book = b
            break
    if not book:
        raise ValueError("Book not found")
    
    book.title = book_data.get('title', book.title)
    book.author = book_data.get('author', book.author)
    book.isbn = book_data.get('isbn', book.isbn)
    book.price = book_data.get('price', book.price)
    
    return book

# Helper function to get a book by ID
def get_book(book_id: str) -> Optional[Book]:
    for b in books:
        if b.id == book_id:
            return b
    return None

# Helper function to get all books with pagination
def get_all_books(page: int, per_page: int) -> List[dict]:
    start = (page - 1) * per_page
    end = start + per_page
    return books[start:end]

# Helper function to get a single book
def get_book_by_id(book_id: str) -> Optional[Book]:
    return get_book(book_id)

# Helper function to create a book
@app.route('/books', methods=['POST'])
def create_book():
    data = request.get_json()
    if not data:
        return jsonify({"error": "No data provided"}), 400
    
    book_data = {
        'title': data.get('title'),
        'author': data.get('author'),
        'isbn': data.get('isbn'),
        'price': data.get('price')
    }
    book = create_book(book_data)
    return jsonify(book), 201

# Helper function to update a book
@app.route('/books/<int:id>', methods=['PUT'])
def update_book(id: int):
    data = request.get_json()
    if not data:
        return jsonify({"error": "No data provided"}), 400
    
    book = get_book_by_id(id)
    if not book:
        return jsonify({"error": "Book not found"}), 404
    
    book_data = {
        'title': data.get('title'),
        'author': data.get('author'),
        'isbn': data.get('isbn'),
        'price': data.get('price')
    }
    
    if book_data:
        book = update_book(id, book_data)
    else:
        return jsonify({"error": "Book not found"}), 404
    
    return jsonify(book), 200

# Helper function to delete a book
@app.route('/books/<int:id>', methods=['DELETE'])
def delete_book(id: int):
    book = get_book_by_id(id)
    if not book:
        return jsonify({"error": "Book not found"}), 404
    
    delete_book(id)
    return jsonify({"message": "Book deleted successfully"}), 200

# Helper function to list all books
@app.route('/books', methods=['GET'])
def list_books():
    page = request.args.get('page', 1, type=int)
    per_page = request.args.get('per_page', 10, type=int)
    
    books = get_all_books(page, per_page)
    return jsonify(books), 200

# Helper function to get a single book
