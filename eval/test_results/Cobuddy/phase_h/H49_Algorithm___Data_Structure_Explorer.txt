**1. LRUCache**

```python
class LRUCache:
    class _Node:
        __slots__ = ("key", "value", "prev", "next")
        def __init__(self, key, value):
            self.key = key
            self.value = value
            self.prev = None
            self.next = None

    def __init__(self, capacity: int):
        self.capacity = capacity
        self.cache = {}                     # key -> _Node
        # dummy head & tail to avoid edge checks
        self.head = self._Node(0, 0)
        self.tail = self._Node(0, 0)
        self.head.next = self.tail
        self.tail.prev = self.head

    def _remove(self, node):
        """Detach node from the linked list."""
        prev_node = node.prev
        next_node = node.next
        prev_node.next = next_node
        next_node.prev = prev_node

    def _add_to_front(self, node):
        """Insert node right after head (most‑recent position)."""
        node.prev = self.head
        node.next = self.head.next
        self.head.next.prev = node
        self.head.next = node

    def get(self, key: int) -> int:
        if key not in self.cache:
            return -1
        node = self.cache[key]
        # mark as recently used
        self._remove(node)
        self._add_to_front(node)
        return node.value

    def put(self, key: int, value: int) -> None:
        if key in self.cache:
            node = self.cache[key]
            node.value = value
            self._remove(node)
            self._add_to_front(node)
        else:
            node = self._Node(key, value)
            self.cache[key] = node
            self._add_to_front(node)
            if len(self.cache) > self.capacity:
                # evict least‑recently used (node before tail)
                lru = self.tail.prev
                self._remove(lru)
                del self.cache[lru.key]
```

**2. MinStack**

```python
class MinStack:
    def __init__(self):
        self.stack = []      # normal values
        self.min_stack = []  # parallel stack of current minima

    def push(self, val: int) -> None:
        self.stack.append(val)
        if not self.min_stack:
            self.min_stack.append(val)
        else:
            self.min_stack.append(min(val, self.min_stack[-1]))

    def pop(self) -> None:
        if self.stack:
            self.stack.pop()
            self.min_stack.pop()

    def top(self) -> int:
        if not self.stack:
            raise IndexError("stack is empty")
        return self.stack[-1]

    def get_min(self) -> int:
        if not self.min_stack:
            raise IndexError("stack is empty")
        return self.min_stack[-1]
```

**3. Running Median**

```python
class _MinHeap:
    """Simple binary min‑heap."""
    def __init__(self):
        self.data = []

    def push(self, val):
        self.data.append(val)
        self._sift_up(len(self.data) - 1)

    def pop(self):
        if not self.data:
            raise IndexError("heap is empty")
        top = self.data[0]
        last = self.data.pop()
        if self.data:
            self.data[0] = last
            self._sift_down(0)
        return top

    def peek(self):
        if not self.data:
            raise IndexError("heap is empty")
        return self.data[0]

    def _sift_up(self, idx):
        data = self.data
        while idx:
            parent = (idx - 1) // 2
            if data[idx] < data[parent]:
                data[idx], data[parent] = data[parent], data[idx]
                idx = parent
            else:
                break

    def _sift_down(self, idx):
        data = self.data
        n = len(data)
        while True:
            left = 2 * idx + 1
            right = 2 * idx + 2
            smallest = idx
            if left < n and data[left] < data[smallest]:
                smallest = left
            if right < n and data[right] < data[smallest]:
                smallest = right
            if smallest != idx:
                data[idx], data[smallest] = data[smallest], data[idx]
                idx = smallest
            else:
                break


def find_median(stream):
    """
    Returns a list of running medians for the given integer stream.
    """
    lower = _MinHeap()   # max‑heap simulated by storing negatives
    higher = _MinHeap()  # normal min‑heap
    medians = []

    for num in stream:
        # Insert into appropriate heap
        if not lower.data or num <= -lower.peek():
            lower.push(-num)
        else:
            higher.push(num)

        # Rebalance so that len(lower) >= len(higher) and difference <= 1
        if len(lower.data) > len(higher.data) + 1:
            moved = -lower.pop()
            higher.push(moved)
        elif len(higher.data) > len(lower.data):
            moved = higher.pop()
            lower.push(-moved)

        # Compute current median
        if len(lower.data) > len(higher.data):
            median = float(-lower.peek())
        else:
            median = (-lower.peek() + higher.peek()) / 2.0
        medians.append(median)

    return medians
```
