```python
from flask import Flask, request, jsonify
from dataclasses import dataclass
from typing import List, Optional
import json

app = Flask(__name__)

@dataclass
class Book:
    id: int
    title: str
    author: str
    isbn: str
    price: float

# In-memory "database"
books: List[Book] = []
next_id = 1

def find_book(book_id: int) -> Optional[Book]:
    return next((b for b in books if b.id == book_id), None)

@app.route('/books', methods=['GET'])
def list_books():
    page = int(request.args.get('page', 1))
    per_page = int(request.args.get('per_page', 10))
    start = (page - 1) * per_page
    end = start + per_page
    paginated_books = books[start:end]
    return jsonify([b.__dict__ for b in paginated_books])

@app.route('/books/<int:book_id>', methods=['GET'])
def get_book(book_id):
    book = find_book(book_id)
    if not book:
        return jsonify({"error": "Book not found"}), 404
    return jsonify(book.__dict__)

@app.route('/books', methods=['POST'])
def create_book():
    global next_id
    data = request.get_json()
    if not data:
        return jsonify({"error": "No data provided"}), 400
    required = ['title', 'author', 'isbn', 'price']
    for field in required:
        if field not in data:
            return jsonify({"error": f"Missing field: {field}"}), 422
    try:
        price = float(data['price'])
    except ValueError:
        return jsonify({"error": "Invalid price format"}), 422
    new_book = Book(id=next_id, title=data['title'], author=data['author'], isbn=data['isbn'], price=price)
    books.append(new_book)
    next_id += 1
    return jsonify(new_book.__dict__), 201

@app.route('/books/<int:book_id>', methods=['PUT'])
def update_book(book_id):
    book = find_book(book_id)
    if not book:
        return jsonify({"error": "Book not found"}), 404
    data = request.get_json()
    if not data:
        return jsonify({"error": "No data provided"}), 400
    if 'title' in data:
        book.title = data['title']
    if 'author' in data:
        book.author = data['author']
    if 'isbn' in data:
        book.isbn = data['isbn']
    if 'price' in data:
        try:
            book.price = float(data['price'])
        except ValueError:
            return jsonify({"error": "Invalid price format"}), 422
    return jsonify(book.__dict__)

@app.route('/books/<int:book_id>', methods=['DELETE'])
def delete_book(book_id):
    global books
    book = find_book(book_id)
    if not book:
        return jsonify({"error": "Book not found"}), 404
    books = [b for b in books if b.id != book_id]
    return '', 204

if __name__ == '__main__':
    app.run(debug=True)
```
