import time
import threading
from collections import defaultdict, deque

class RateLimiter:
    def __init__(self, max_requests: int, window_seconds: float, burst: int = 0):
        self.max_requests = max_requests
        self.window_seconds = window_seconds
        self.burst = burst
        self.burst_window = 2 * window_seconds if burst > 0 else window_seconds
        self.data = defaultdict(lambda: {'timestamps': deque(), 'burst_timestamps': deque()})
        self.lock = threading.Lock()

    def allow(self, user_id: str, endpoint: str = "default") -> dict:
        key = (user_id, endpoint)
        current_time = time.time()
        with self.lock:
            self._cleanup_entry(key, current_time)
            entry = self.data[key]
            window_count = self._count_in_window(entry['timestamps'], current_time, self.window_seconds)
            burst_count = self._count_in_window(entry['burst_timestamps'], current_time, self.burst_window)

            if window_count < self.max_requests and burst_count < self.burst:
                entry['timestamps'].append(current_time)
                if self.burst > 0:
                    entry['burst_timestamps'].append(current_time)
                remaining = self.max_requests - window_count - 1
                return {"allowed": True, "remaining": remaining, "retry_after": None}
            else:
                oldest = entry['timestamps'][0] if entry['timestamps'] else current_time
                retry_after = max(0, oldest + self.window_seconds - current_time)
                remaining = max(0, self.max_requests - window_count)
                return {"allowed": False, "remaining": remaining, "retry_after": retry_after}

    def allow_priority(self, user_id: str, endpoint: str = "default") -> dict:
        return {"allowed": True, "remaining": None, "retry_after": None}

    def cleanup(self, max_age_seconds: float = None) -> int:
        if max_age_seconds is None:
            max_age_seconds = 2 * self.window_seconds
        current_time = time.time()
        removed = 0
        with self.lock:
            for key in list(self.data.keys()):
                entry = self.data[key]
                initial_len = len(entry['timestamps']) + len(entry['burst_timestamps'])
                while entry['timestamps'] and entry['timestamps'][0] < current_time - max_age_seconds:
                    entry['timestamps'].popleft()
                while entry['burst_timestamps'] and entry['burst_timestamps'][0] < current_time - max_age_seconds:
                    entry['burst_timestamps'].popleft()
                final_len = len(entry['timestamps']) + len(entry['burst_timestamps'])
                removed += initial_len - final_len
                if not entry['timestamps'] and not entry['burst_timestamps']:
                    del self.data[key]
        return removed

    def get_usage(self, user_id: str, endpoint: str = "default") -> dict:
        key = (user_id, endpoint)
        current_time = time.time()
        with self.lock:
            self._cleanup_entry(key, current_time)
            entry = self.data[key]
            window_count = self._count_in_window(entry['timestamps'], current_time, self.window_seconds)
            burst_count = self._count_in_window(entry['burst_timestamps'], current_time, self.burst_window)
            burst_remaining = self.burst - burst_count
            return {
                "used": window_count,
                "limit": self.max_requests,
                "window_seconds": self.window_seconds,
                "burst_remaining": burst_remaining
            }

    def to_dict(self) -> dict:
        with self.lock:
            return {
                'max_requests': self.max_requests,
                'window_seconds': self.window_seconds,
                'burst': self.burst,
                'burst_window': self.burst_window,
                'data': {
                    str(k): {
                        'timestamps': list(self.data[k]['timestamps']),
                        'burst_timestamps': list(self.data[k]['burst_timestamps'])
                    } for k in self.data
                }
            }

    @classmethod
    def from_dict(cls, data: dict):
        instance = cls(data['max_requests'], data['window_seconds'], data['burst'])
        instance.burst_window = data['burst_window']
        instance.data = defaultdict(lambda: {'timestamps': deque(), 'burst_timestamps': deque()})
        for k, v in data['data'].items():
            user_id, endpoint = eval(k)
            instance.data[(user_id, endpoint)]['timestamps'] = deque(v['timestamps'])
            instance.data[(user_id, endpoint)]['burst_timestamps'] = deque(v['burst_timestamps'])
        return instance

    def _cleanup_entry(self, key, current_time):
        entry = self.data[key]
        while entry['timestamps'] and entry['timestamps'][0] < current_time - self.window_seconds:
            entry['timestamps'].popleft()
        while entry['burst_timestamps'] and entry['burst_timestamps'][0] < current_time - self.burst_window:
            entry['burst_timestamps'].popleft()

    def _count_in_window(self, timestamps, current_time, window):
        cutoff = current_time - window
        count = 0
        for ts in reversed(timestamps):
            if ts >= cutoff:
                count += 1
            else:
                break
        return count
