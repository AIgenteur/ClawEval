from flask import Flask, request, jsonify
from dataclasses import dataclass
from typing import List, Optional
import uuid

app = Flask(__name__)

# In-memory database
books: List[dict] = []

@dataclass
class Book:
    id: str
    title: str
    author: str
    isbn: str
    price: float

    def to_dict(self) -> dict:
        return {
            "id": self.id,
            "title": self.title,
            "author": self.author,
            "isbn": self.isbn,
            "price": self.price
        }

def validate_book_data(data: dict) -> Optional[dict]:
    """Validates required fields and basic types."""
    required_fields = ["title", "author", "isbn", "price"]
    
    for field in required_fields:
        if field not in data or not data[field]:
            return None
    
    if not isinstance(data["title"], str) or not data["title"].strip():
        return None
    if not isinstance(data["author"], str) or not data["author"].strip():
        return None
    if not isinstance(data["isbn"], str) or not data["isbn"].strip():
        return None
    if not isinstance(data["price"], (int, float)) or data["price"] < 0:
        return None
    
    return data

@app.route("/books", methods=["GET"])
def get_books():
    page = request.args.get("page", 1, type=int)
    per_page = request.args.get("per_page", 10, type=int)
    
    if per_page < 1:
        per_page = 10
    if per_page > 100:
        per_page = 100
    
    start = (page - 1) * per_page
    end = start + per_page
    
    paginated_books = books[start:end]
    
    return jsonify({
        "books": [book.to_dict() for book in paginated_books],
        "total": len(books),
        "page": page,
        "per_page": per_page,
        "total_pages": (len(books) + per_page - 1) // per_page
    }), 200

@app.route("/books/<int:book_id>", methods=["GET"])
def get_book(book_id: int):
    # In a real scenario, IDs might be UUIDs or strings. 
    # Here we assume integer IDs for simplicity based on the prompt structure.
    # If the DB uses UUIDs, this route would need to be /books/<uuid>
    for book in books:
        if book["id"] == book_id:
            return jsonify(book.to_dict()), 200
    
    return jsonify({"error": "Book not found"}), 404

@app.route("/books", methods=["POST"])
def create_book():
    data = request.get_json()
    
    if not validate_book_data(data):
        return jsonify({"error": "Missing required fields or invalid data"}), 400
    
    new_book = Book(
        id=str(uuid.uuid4()),
        title=data["title"],
        author=data["author"],
        isbn=data["isbn"],
        price=float(data["price"])
    )
    
    books.append(new_book)
    
    return jsonify(new_book.to_dict()), 201

@app.route("/books/<int:book_id>", methods=["PUT"])
def update_book(book_id: int):
    book = next((b for b in books if b["id"] == book_id), None)
    
    if not book:
        return jsonify({"error": "Book not found"}), 404
    
    data = request.get_json()
    
    if not validate_book_data(data):
        return jsonify({"error": "Missing required fields or invalid data"}), 400
    
    # Update only provided fields
    book["title"] = data.get("title", book["title"])
    book["author"] = data.get("author", book["author"])
    book["isbn"] = data.get("isbn", book["isbn"])
    book["price"] = float(data.get("price", book["price"]))
    
    return jsonify(book.to_dict()), 200

@app.route("/books/<int:book_id>", methods=["DELETE"])
def delete_book(book_id: int):
    global books
    original_len = len(books)
    
    books = [b for b in books if b["id"] != book_id]
    
    if len(books) == original_len:
        return jsonify({"error": "Book not found"}), 404
    
    return jsonify({"message": "Book deleted successfully"}), 200

if __name__ == "__main__":
    app.run(debug=True)
