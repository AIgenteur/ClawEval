```python
from flask import Flask, jsonify, request
from dataclasses import dataclass, asdict
from typing import List, Optional, Tuple, Any

app = Flask(__name__)

@dataclass
class Book:
    id: int
    title: str
    author: str
    isbn: str
    price: float

books_db: List[Book] = []
next_id = 1

def make_error(message: str, status_code: int) -> Tuple[Any, int]:
    return jsonify({"error": message}), status_code

def validate_book_data(data: dict, is_update: bool = False) -> Optional[Tuple[Any, int]]:
    required_fields = ['title', 'author', 'isbn', 'price']
    
    if not is_update:
        for field in required_fields:
            if field not in data or data[field] is None:
                return make_error(f"Missing required field: {field}", 400)
    
    if 'price' in data and data['price'] is not None:
        try:
            price = float(data['price'])
            if price < 0:
                return make_error("Price must be non-negative", 422)
        except (TypeError, ValueError):
            return make_error("Price must be a valid number", 422)
            
    return None

@app.route('/books', methods=['GET'])
def get_books():
    try:
        page = int(request.args.get('page', 1))
        per_page = int(request.args.get('per_page', 10))
    except ValueError:
        return make_error("Invalid pagination parameters", 400)

    if page < 1 or per_page < 1:
        return make_error("Page and per_page must be positive integers", 400)

    start = (page - 1) * per_page
    end = start + per_page
    paginated_books = books_db[start:end]

    return jsonify({
        "books": [asdict(b) for b in paginated_books],
        "page": page,
        "per_page": per_page,
        "total": len(books_db)
    }), 200

@app.route('/books/<int:book_id>', methods=['GET'])
def get_book(book_id):
    book = next((b for b in books_db if b.id == book_id), None)
    if not book:
        return make_error("Book not found", 404)
    return jsonify(asdict(book)), 200

@app.route('/books', methods=['POST'])
def create_book():
    global next_id
    if not request.json:
        return make_error("Request body must be JSON", 400)

    validation_error = validate_book_data(request.json, is_update=False)
    if validation_error:
        return validation_error

    new_book = Book(
        id=next_id,
        title=request.json['title'],
        author=request.json['author'],
        isbn=request.json['isbn'],
        price=float(request.json['price'])
    )
    books_db.append(new_book)
    next_id += 1

    return jsonify(asdict(new_book)), 201

@app.route('/books/<int:book_id>', methods=['PUT'])
def update_book(book_id):
    book = next((b for b in books_db if b.id == book_id), None)
    if not book:
        return make_error("Book not found", 404)

    if not request.json:
        return make_error("Request body must be JSON", 400)

    validation_error = validate_book_data(request.json, is_update=True)
    if validation_error:
        return validation_error

    data = request.json
    if 'title' in data:
        book.title = data['title']
    if 'author' in data:
        book.author = data['author']
    if 'isbn' in data:
        book.isbn = data['isbn']
    if 'price' in data:
        book.price = float(data['price'])

    return jsonify(asdict(book)), 200

@app.route('/books/<int:book_id>', methods=['DELETE'])
def delete_book(book_id):
    global books_db
    book = next((b for b in books_db if b.id == book_id), None)
    if not book:
        return make_error("Book not found", 404)

    books_db = [b for b in books_db if b.id != book_id]
    return jsonify({"message": "Book deleted successfully"}), 200

if __name__ == '__main__':
    app.run(debug=True)
```
