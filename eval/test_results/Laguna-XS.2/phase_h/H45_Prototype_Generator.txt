
```python
class JSONValidator:
    def validate(self, data, schema):
        errors = []
        
        for field_name, field_schema in schema.items():
            # Check if required field is missing
            if field_schema.get("required", False) and field_name not in data:
                errors.append(f"Field '{field_name}' is required but missing")
                continue
            
            # Skip validation if field is not present and not required
            if field_name not in data:
                continue
            
            value = data[field_name]
            
            # Check type
            expected_type = field_schema.get("type")
            if expected_type and not self._check_type(value, expected_type):
                actual_type = type(value).__name__
                errors.append(f"Field '{field_name}' has wrong type. Expected {expected_type}, got {actual_type}")
                continue
            
            # Validate numbers (int/float) with min/max
            if expected_type in ("int", "float"):
                if "min" in field_schema and value < field_schema["min"]:
                    errors.append(f"Field '{field_name}' is less than minimum {field_schema['min']}")
                if "max" in field_schema and value > field_schema["max"]:
                    errors.append(f"Field '{field_name}' is greater than maximum {field_schema['max']}")
            
            # Validate strings with min_length/max_length
            if expected_type == "str":
                if "min_length" in field_schema and len(value) < field_schema["min_length"]:
                    errors.append(f"Field '{field_name}' is shorter than minimum length {field_schema['min_length']}")
                if "max_length" in field_schema and len(value) > field_schema["max_length"]:
                    errors.append(f"Field '{field_name}' is longer than maximum length {field_schema['max_length']}")
            
            # Validate lists with min_length/max_length
            if expected_type == "list":
                if "min_length" in field_schema and len(value) < field_schema["min_length"]:
                    errors.append(f"Field '{field_name}' has fewer than minimum {field_schema['min_length']} items")
                if "max_length" in field_schema and len(value) > field_schema["max_length"]:
                    errors.append(f"Field '{field_name}' has more than maximum {field_schema['max_length']} items")
        
        return {"valid": len(errors) == 0, "errors": errors}
    
    def _check_type(self, value, expected_type):
        if expected_type == "str":
            return isinstance(value, str)
        elif expected_type == "int":
            return isinstance(value, int) and not isinstance(value, bool)
        elif expected_type == "float":
            return isinstance(value, (float, int)) and not isinstance(value, bool)
        elif expected_type == "bool":
            return isinstance(value, bool)
        elif expected_type == "list":
            return isinstance(value, list)
        elif expected_type == "dict":
            return isinstance(value, dict)
        return True
```

**Usage Example:**

```python
validator = JSONValidator()

schema = {
    "name": {"type": "str", "required": True, "min_length": 1, "max_length": 50},
    "age": {"type": "int", "required": True, "min": 0, "max": 150},
    "tags": {"type": "list", "required": False, "min_length": 1, "max_length": 10}
}

data = {"name": "", "age": 200, "tags": []}
result = validator.validate(data, schema)
print(result)
# Output: {'valid': False, 'errors': [
#     "Field 'name' is shorter than minimum length 1",
#     "Field 'age' is greater than maximum 150",
#     "Field 'tags' has fewer than minimum 1 items"
# ]}
```
