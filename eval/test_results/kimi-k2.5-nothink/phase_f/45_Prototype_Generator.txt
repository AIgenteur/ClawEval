```python
from flask import Flask, request, jsonify
from dataclasses import dataclass, asdict
from typing import Optional
import re

app = Flask(__name__)


@dataclass
class Book:
    id: int
    title: str
    author: str
    isbn: str
    price: float


books_db: list[Book] = []
next_id = 1


def validate_book_data(data: dict, required_fields: bool = True) -> tuple[bool, Optional[str]]:
    if not isinstance(data, dict):
        return False, "Request body must be a JSON object"
    
    fields = ['title', 'author', 'isbn', 'price']
    
    for field in fields:
        if required_fields and field not in data:
            return False, f"Missing required field: {field}"
        
        if field in data:
            if field == 'price':
                if not isinstance(data[field], (int, float)):
                    return False, "Price must be a number"
                if data[field] < 0:
                    return False, "Price cannot be negative"
            elif field in ['title', 'author', 'isbn']:
                if not isinstance(data[field], str):
                    return False, f"{field} must be a string"
                if not data[field].strip():
                    return False, f"{field} cannot be empty"
    
    if 'isbn' in data:
        isbn_pattern = r'^(?:\d{9}[\dXx]|\d{13})$'
        clean_isbn = data['isbn'].replace('-', '').replace(' ', '')
        if not re.match(isbn_pattern, clean_isbn):
            return False, "Invalid ISBN format"
    
    return True, None


@app.route('/books', methods=['GET'])
def list_books():
    page = request.args.get('page', 1, type=int)
    per_page = request.args.get('per_page', 10, type=int)
    
    if page < 1:
        return jsonify({"error": "Page must be at least 1"}), 400
    if per_page < 1 or per_page > 100:
        return jsonify({"error": "Per page must be between 1 and 100"}), 400
    
    total = len(books_db)
    start = (page - 1) * per_page
    end = start + per_page
    
    paginated_books = books_db[start:end]
    
    return jsonify({
        "books": [asdict(book) for book in paginated_books],
        "total": total,
        "page": page,
        "per_page": per_page,
        "pages": (total + per_page - 1) // per_page
    }), 200


@app.route('/books/<int:book_id>', methods=['GET'])
def get_book(book_id):
    book = next((b for b in books_db if b.id == book_id), None)
    if book is None:
        return jsonify({"error": f"Book with id {book_id} not found"}), 404
    return jsonify(asdict(book)), 200


@app.route('/books', methods=['POST'])
def create_book():
    global next_id
    
    if not request.is_json:
        return jsonify({"error": "Content-Type must be application/json"}), 400
    
    data = request.get_json()
    
    valid, error_msg = validate_book_data(data)
    if not valid:
        return jsonify({"error": error_msg}), 422
    
    existing = next((b for b in books_db if b.isbn == data['isbn']), None)
    if existing:
        return jsonify({"error": f"Book with ISBN {data['isbn']} already exists"}), 422
    
    book = Book(
        id=next_id,
        title=data['title'].strip(),
        author=data['author'].strip(),
        isbn=data['isbn'].replace('-', '').replace(' ', ''),
        price=float(data['price'])
    )
    
    books_db.append(book)
    next_id += 1
    
    return jsonify(asdict(book)), 201


@app.route('/books/<int:book_id>', methods=['PUT'])
def update_book(book_id):
    book = next((b for b in books_db if b.id == book_id), None)
    if book is None:
        return jsonify({"error": f"Book with id {book_id} not found"}), 404
    
    if not request.is_json:
        return jsonify({"error": "Content-Type must be application/json"}), 400
    
    data = request.get_json()
    
    valid, error_msg = validate_book_data(data, required_fields=False)
    if not valid:
        return jsonify({"error": error_msg}), 422
    
    if 'isbn' in data:
        clean_isbn = data['isbn'].replace('-', '').replace(' ', '')
        existing = next((b for b in books_db if b.isbn == clean_isbn and b.id != book_id), None)
        if existing:
            return jsonify({"error": f"Book with ISBN {clean_isbn} already exists"}), 422
    
    if 'title' in data:
        book.title = data['title'].strip()
    if 'author' in data:
        book.author = data['author'].strip()
    if 'isbn' in data:
        book.isbn = data['isbn'].replace('-', '').replace(' ', '')
    if 'price' in data:
        book.price = float(data['price'])
    
    return jsonify(asdict(book)), 200


@app.route('/books/<int:book_id>', methods=['DELETE'])
def delete_book(book_id):
    global books_db
    
    book = next((b for b in books_db if b.id == book_id), None)
    if book is None:
        return jsonify({"error": f"Book with id {book_id} not found"}), 404
    
    books_db = [b for b in books_db if b.id != book_id]
    
    return '', 204


@app.errorhandler(404)
def not_found(error):
    return jsonify({"error": "Resource not found"}), 404


@app.errorhandler(405)
def method_not_allowed(error):
    return jsonify({"error": "Method not allowed"}), 405


if __name__ == '__main__':
    app.run(debug=True)
```
