import time
import threading
from collections import defaultdict, deque

class RateLimiter:
    def __init__(self, max_requests: int, window_seconds: float, burst: int = 0):
        self.max_requests = max_requests
        self.window_seconds = window_seconds
        self.burst = burst
        self.state = defaultdict(lambda: {
            'tokens': 0.0,
            'last_refill': 0.0,
            'burst_tokens': 0.0,
            'last_burst_refill': 0.0,
            'history': deque()
        })
        self.lock = threading.Lock()

    def allow(self, user_id: str, endpoint: str = "default") -> dict:
        with self.lock:
            key = (user_id, endpoint)
            entry = self.state[key]
            now = time.time()

            # Refill tokens
            elapsed = now - entry['last_refill']
            refill = elapsed * (self.max_requests / self.window_seconds)
            entry['tokens'] = min(self.max_requests, entry['tokens'] + refill)
            entry['last_refill'] = now

            # Refill burst tokens (2x slower)
            burst_elapsed = now - entry['last_burst_refill']
            burst_refill = burst_elapsed * (self.burst / (2 * self.window_seconds))
            entry['burst_tokens'] = min(self.burst, entry['burst_tokens'] + burst_refill)
            entry['last_burst_refill'] = now

            # Check if allowed
            total_tokens = entry['tokens'] + entry['burst_tokens']
            if total_tokens >= 1:
                entry['tokens'] -= 1
                entry['history'].append(now)
                return {
                    "allowed": True,
                    "remaining": max(0, int(self.max_requests - (entry['tokens'] + entry['burst_tokens']))),
                    "retry_after": None
                }
            else:
                retry_after = (1 - total_tokens) * (self.window_seconds / self.max_requests)
                return {
                    "allowed": False,
                    "remaining": 0,
                    "retry_after": retry_after
                }

    def allow_priority(self, user_id: str, endpoint: str = "default") -> dict:
        return {"allowed": True, "remaining": 0, "retry_after": None}

    def cleanup(self, max_age_seconds: float = None) -> int:
        if max_age_seconds is None:
            max_age_seconds = 2 * self.window_seconds
        cutoff = time.time() - max_age_seconds
        removed = 0
        with self.lock:
            for key in list(self.state.keys()):
                entry = self.state[key]
                while entry['history'] and entry['history'][0] < cutoff:
                    entry['history'].popleft()
                    removed += 1
                if not entry['history']:
                    del self.state[key]
        return removed

    def get_usage(self, user_id: str, endpoint: str = "default") -> dict:
        with self.lock:
            key = (user_id, endpoint)
            entry = self.state.get(key, {
                'tokens': 0.0,
                'last_refill': 0.0,
                'burst_tokens': 0.0,
                'last_burst_refill': 0.0,
                'history': deque()
            })
            now = time.time()
            elapsed = now - entry['last_refill']
            refill = elapsed * (self.max_requests / self.window_seconds)
            tokens = min(self.max_requests, entry['tokens'] + refill)
            burst_elapsed = now - entry['last_burst_refill']
            burst_refill = burst_elapsed * (self.burst / (2 * self.window_seconds))
            burst_tokens = min(self.burst, entry['burst_tokens'] + burst_refill)
            return {
                "used": len(entry['history']),
                "limit": self.max_requests + self.burst,
                "window_seconds": self.window_seconds,
                "burst_remaining": int(self.burst - (entry['burst_tokens'] + burst_refill))
            }

    def to_dict(self) -> dict:
        with self.lock:
            return {
                "max_requests": self.max_requests,
                "window_seconds": self.window_seconds,
                "burst": self.burst,
                "state": {
                    str(key): {
                        "tokens": entry['tokens'],
                        "last_refill": entry['last_refill'],
                        "burst_tokens": entry['burst_tokens'],
                        "last_burst_refill": entry['last_burst_refill'],
                        "history": list(entry['history'])
                    }
                    for key, entry in self.state.items()
                }
            }

    @classmethod
    def from_dict(cls, data: dict) -> 'RateLimiter':
        limiter = cls(
            max_requests=data["max_requests"],
            window_seconds=data["window_seconds"],
            burst=data["burst"]
        )
        with limiter.lock:
            limiter.state = defaultdict(lambda: {
                'tokens': 0.0,
                'last_refill': 0.0,
                'burst_tokens': 0.0,
                'last_burst_refill': 0.0,
                'history': deque()
            })
            for key_str, entry_data in data["state"].items():
                key = eval(key_str)
                entry = limiter.state[key]
                entry['tokens'] = entry_data['tokens']
                entry['last_refill'] = entry_data['last_refill']
                entry['burst_tokens'] = entry_data['burst_tokens']
                entry['last_burst_refill'] = entry_data['last_burst_refill']
                entry['history'] = deque(entry_data['history'])
        return limiter
