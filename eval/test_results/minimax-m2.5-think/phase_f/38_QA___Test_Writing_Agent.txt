import pytest

# === Valid Inputs ===

def test_zero_income_single():
    tax, rate = calculate_tax(0, 'single')
    assert tax == 0
    assert rate == 0


def test_zero_income_married():
    tax, rate = calculate_tax(0, 'married')
    assert tax == 0
    assert rate == 0


def test_typical_income_single():
    """Income in 12% bracket"""
    tax, rate = calculate_tax(30000, 'single')
    expected_tax = 11600 * 0.10 + (30000 - 11600) * 0.12
    assert tax == round(expected_tax, 2)
    assert rate == round(expected_tax / 30000 * 100, 2)


def test_typical_income_married():
    """Income in 12% bracket"""
    tax, rate = calculate_tax(50000, 'married')
    expected_tax = 23200 * 0.10 + (50000 - 23200) * 0.12
    assert tax == round(expected_tax, 2)
    assert rate == round(expected_tax / 50000 * 100, 2)

    """Income in top 37% bracket"""
    tax, rate = calculate_tax(1000000, 'single')

def test_high_income_married():
    """Income in top 37% bracket"""
    tax, rate = calculate_tax(1000000, 'married')
    assert rate == 37.0


def test_single_vs_married_same_income():
    """Married couples get twice the standard deduction effectively"""
    tax_single, rate_single = calculate_tax(40000, 'single')
    tax_married, rate_married = calculate_tax(40000, 'married')
    # Married should pay less tax at same income
    assert tax_married < tax_single
