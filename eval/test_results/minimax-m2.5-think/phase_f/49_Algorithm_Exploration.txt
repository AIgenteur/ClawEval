from collections import OrderedDict

def lru_cache(capacity):
    """Return an LRU cache with the specified capacity."""
    class LRU:
        def __init__(self, capacity):
            self.capacity = capacity
            self.cache = OrderedDict()  # key -> value, ordered by access time

        def get(self, key):
            """Return the value for key if present, else -1."""
            if key not in self.cache:
                return -1
            # Move to the end to mark it as recently used.
            self.cache.move_to_end(key)
            return self.cache[key]

        def put(self, key, value):
            """Insert or update key with value. Evict LRU item if over capacity."""
            if key in self.cache:
                # Update existing key and mark it as recently used.
                self.cache.move_to_end(key)
            self.cache[key] = value
            # If we exceed capacity, remove the least recently used item.
            if len(self.cache) > self.capacity:
                self.cache.popitem(last=False)  # pop the first (oldest) item

    if capacity <= 0:
        raise ValueError("Capacity must be a positive integer")
    return LRU(capacity)
