import re

def parse_duration(s: str) -> int:
    if not s:
        return -1
    pattern = re.compile(r'(\d+)([hms])')
    pos = 0
    total = 0
    seen = set()
    for m in pattern.finditer(s):
        if m.start() != pos:          # gap or unexpected character
            return -1
        value = int(m.group(1))
        unit = m.group(2)
        if unit in seen:              # duplicate unit
            return -1
        seen.add(unit)
        if unit == 'h':
            total += value * 3600
        elif unit == 'm':
            total += value * 60
        else:  # 's'
            total += value
        pos = m.end()
    return total if pos == len(s) else -1
