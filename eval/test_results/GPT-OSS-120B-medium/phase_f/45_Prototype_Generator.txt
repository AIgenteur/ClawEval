from dataclasses import dataclass, asdict
from flask import Flask, request, jsonify, abort

app = Flask(__name__)

@dataclass
class Book:
    id: int
    title: str
    author: str
    isbn: str
    price: float

# In‑memory "database"
books: list[Book] = []
_next_id = 1

def _find_book(book_id: int) -> Book | None:
    return next((b for b in books if b.id == book_id), None)

def _validate_book_payload(data: dict, partial: bool = False):
    required = {"title", "author", "isbn", "price"}
    if not partial:
        missing = required - data.keys()
        if missing:
            abort(422, description=f"Missing fields: {', '.join(missing)}")
    else:
        # ignore unknown fields on partial update
        data = {k: v for k, v in data.items() if k in required}
    # type checks
    if "price" in data:
        try:
            data["price"] = float(data["price"])
        except (TypeError, ValueError):
            abort(422, description="Price must be a number")
    return data

@app.errorhandler(400)
@app.errorhandler(404)
@app.errorhandler(422)
def handle_error(err):
    response = jsonify(error=err.description if err.description else err.name)
    response.status_code = err.code if hasattr(err, "code") else 500
    return response

@app.route("/books", methods=["GET"])
def list_books():
    page = int(request.args.get("page", 1))
    per_page = int(request.args.get("per_page", 10))
    start = (page - 1) * per_page
    end = start + per_page
    items = [asdict(b) for b in books[start:end]]
    return jsonify(items), 200

@app.route("/books/<int:book_id>", methods=["GET"])
def get_book(book_id):
    book = _find_book(book_id)
    if not book:
        abort(404, description="Book not found")
    return jsonify(asdict(book)), 200

@app.route("/books", methods=["POST"])
def create_book():
    if not request.is_json:
        abort(400, description="Request must be JSON")
    data = request.get_json()
    data = _validate_book_payload(data)
    global _next_id
    book = Book(
        id=_next_id,
        title=data["title"],
        author=data["author"],
        isbn=data["isbn"],
        price=data["price"],
    )
    books.append(book)
    _next_id += 1
    return jsonify(asdict(book)), 201

@app.route("/books/<int:book_id>", methods=["PUT"])
def update_book(book_id):
    if not request.is_json:
        abort(400, description="Request must be JSON")
    book = _find_book(book_id)
    if not book:
        abort(404, description="Book not found")
    data = request.get_json()
    data = _validate_book_payload(data, partial=False)
    book.title = data["title"]
    book.author = data["author"]
    book.isbn = data["isbn"]
    book.price = data["price"]
    return jsonify(asdict(book)), 200

@app.route("/books/<int:book_id>", methods=["DELETE"])
def delete_book(book_id):
    book = _find_book(book_id)
    if not book:
        abort(404, description="Book not found")
    books.remove(book)
    return "", 204

if __name__ == "__main__":
    app.run(debug=True)
