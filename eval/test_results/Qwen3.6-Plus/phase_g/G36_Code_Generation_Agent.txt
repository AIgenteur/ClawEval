import time
import threading

class RateLimiter:
    def __init__(self, max_requests: int, window_seconds: float, burst: int = 0):
        self.max_requests = max_requests
        self.window_seconds = window_seconds
        self.burst = burst
        self._lock = threading.Lock()
        self._buckets = {}

    def allow(self, user_id: str, endpoint: str = "default") -> dict:
        with self._lock:
            now = time.time()
            key = (user_id, endpoint)
            if key not in self._buckets:
                self._buckets[key] = {
                    "timestamps": [],
                    "burst_tokens": float(self.burst),
                    "last_burst_update": now
                }

            bucket = self._buckets[key]
            cutoff = now - self.window_seconds
            bucket["timestamps"] = [t for t in bucket["timestamps"] if t > cutoff]

            if self.burst > 0:
                elapsed = now - bucket["last_burst_update"]
                regen_rate = self.burst / (2 * self.window_seconds)
                bucket["burst_tokens"] = min(float(self.burst), bucket["burst_tokens"] + elapsed * regen_rate)
                bucket["last_burst_update"] = now

            current_count = len(bucket["timestamps"])
            allowed = False
            retry_after = None

            if current_count < self.max_requests:
                allowed = True
                bucket["timestamps"].append(now)
            elif self.burst > 0 and bucket["burst_tokens"] >= 1.0:
                allowed = True
                bucket["timestamps"].append(now)
                bucket["burst_tokens"] -= 1.0
            else:
                window_retry = (bucket["timestamps"][0] + self.window_seconds - now) if bucket["timestamps"] else float('inf')
                burst_retry = float('inf')
                if self.burst > 0:
                    regen_rate = self.burst / (2 * self.window_seconds)
                    if regen_rate > 0:
                        burst_retry = (1.0 - bucket["burst_tokens"]) / regen_rate
                retry_after = min(window_retry, burst_retry)
                if retry_after == float('inf'):
                    retry_after = None
                else:
                    retry_after = max(0.0, retry_after)

            remaining = max(0, self.max_requests - len(bucket["timestamps"]))
            return {"allowed": allowed, "remaining": remaining, "retry_after": retry_after}

    def allow_priority(self, user_id: str, endpoint: str = "default") -> dict:
        return {"allowed": True, "remaining": self.max_requests, "retry_after": None}

    def cleanup(self, max_age_seconds: float = None) -> int:
        if max_age_seconds is None:
            max_age_seconds = 2 * self.window_seconds
        with self._lock:
            now = time.time()
            cutoff = now - max_age_seconds
            to_remove = []
            for key, bucket in self._buckets.items():
                if not bucket["timestamps"] or max(bucket["timestamps"]) <= cutoff:
                    to_remove.append(key)
            for key in to_remove:
                del self._buckets[key]
            return len(to_remove)

    def get_usage(self, user_id: str, endpoint: str = "default") -> dict:
        with self._lock:
            now = time.time()
            key = (user_id, endpoint)
            if key not in self._buckets:
                return {"used": 0, "limit": self.max_requests, "window_seconds": self.window_seconds, "burst_remaining": self.burst}

            bucket = self._buckets[key]
            cutoff = now - self.window_seconds
            timestamps = [t for t in bucket["timestamps"] if t > cutoff]

            if self.burst > 0:
                elapsed = now - bucket["last_burst_update"]
                regen_rate = self.burst / (2 * self.window_seconds)
                burst_rem = min(float(self.burst), bucket["burst_tokens"] + elapsed * regen_rate)
            else:
                burst_rem = 0.0

            return {
                "used": len(timestamps),
                "limit": self.max_requests,
                "window_seconds": self.window_seconds,
                "burst_remaining": int(burst_rem)
            }

    def to_dict(self) -> dict:
        with self._lock:
            return {
                "max_requests": self.max_requests,
                "window_seconds": self.window_seconds,
                "burst": self.burst,
                "buckets": {k: v for k, v in self._buckets.items()}
            }

    @classmethod
    def from_dict(cls, data: dict) -> 'RateLimiter':
        limiter = cls(data["max_requests"], data["window_seconds"], data.get("burst", 0))
        limiter._buckets = data.get("buckets", {})
        return limiter
