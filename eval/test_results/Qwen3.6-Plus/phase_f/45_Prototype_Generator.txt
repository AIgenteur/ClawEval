from flask import Flask, request, jsonify
from dataclasses import dataclass, asdict
from typing import List

app = Flask(__name__)

@dataclass
class Book:
    id: int
    title: str
    author: str
    isbn: str
    price: float

books_db: List[Book] = []
_next_id = 1

def validate_book_data(data: dict, require_all: bool = True):
    if not isinstance(data, dict):
        return None, ({"error": "Invalid JSON payload"}, 400)

    required_fields = ["title", "author", "isbn", "price"]
    if require_all:
        missing = [f for f in required_fields if f not in data]
        if missing:
            return None, ({"error": f"Missing required fields: {', '.join(missing)}"}, 400)

    if not isinstance(data.get("title"), str) or not data["title"].strip():
        return None, ({"error": "title must be a non-empty string"}, 422)
    if not isinstance(data.get("author"), str) or not data["author"].strip():
        return None, ({"error": "author must be a non-empty string"}, 422)
    if not isinstance(data.get("isbn"), str) or not data["isbn"].strip():
        return None, ({"error": "isbn must be a non-empty string"}, 422)
    if not isinstance(data.get("price"), (int, float)) or data["price"] < 0:
        return None, ({"error": "price must be a non-negative number"}, 422)

    return data, None

@app.route("/books", methods=["GET"])
def list_books():
    page = request.args.get("page", 1, type=int)
    per_page = request.args.get("per_page", 10, type=int)

    if page < 1 or per_page < 1:
        return jsonify({"error": "page and per_page must be positive integers"}), 400

    start = (page - 1) * per_page
    end = start + per_page
    return jsonify([asdict(b) for b in books_db[start:end]]), 200

@app.route("/books/<int:book_id>", methods=["GET"])
def get_book(book_id):
    book = next((b for b in books_db if b.id == book_id), None)
    if not book:
        return jsonify({"error": "Book not found"}), 404
    return jsonify(asdict(book)), 200

@app.route("/books", methods=["POST"])
def create_book():
    global _next_id
    data, error = validate_book_data(request.get_json(silent=True))
    if error:
        return jsonify(error[0]), error[1]

    new_book = Book(
        id=_next_id,
        title=data["title"],
        author=data["author"],
        isbn=data["isbn"],
        price=float(data["price"])
    )
    books_db.append(new_book)
    _next_id += 1
    return jsonify(asdict(new_book)), 201

@app.route("/books/<int:book_id>", methods=["PUT"])
def update_book(book_id):
    book = next((b for b in books_db if b.id == book_id), None)
    if not book:
        return jsonify({"error": "Book not found"}), 404

    data, error = validate_book_data(request.get_json(silent=True), require_all=False)
    if error:
        return jsonify(error[0]), error[1]

    for field in ["title", "author", "isbn", "price"]:
        if field in data:
            setattr(book, field, float(data[field]) if field == "price" else data[field])

    return jsonify(asdict(book)), 200

@app.route("/books/<int:book_id>", methods=["DELETE"])
def delete_book(book_id):
    book = next((b for b in books_db if b.id == book_id), None)
    if not book:
        return jsonify({"error": "Book not found"}), 404

    books_db.remove(book)
    return jsonify({"message": "Book deleted successfully"}), 200

if __name__ == "__main__":
    app.run(debug=True)
