class _Node:
    """Doubly linked list node for LRUCache."""
    __slots__ = ('key', 'val', 'prev', 'next')
    def __init__(self, key, val):
        self.key = key
        self.val = val
        self.prev = None
        self.next = None

class LRUCache:
    """Least Recently Used Cache with O(1) get/put."""
    def __init__(self, capacity: int):
        self.capacity = capacity
        self.cache = {}
        # sentinel head and tail
        self.head = _Node(0, 0)
        self.tail = _Node(0, 0)
        self.head.next = self.tail
        self.tail.prev = self.head

    def _remove(self, node: _Node) -> None:
        p, n = node.prev, node.next
        p.next = n
        n.prev = p

    def _add_to_front(self, node: _Node) -> None:
        """Insert right after head."""
        nxt = self.head.next
        self.head.next = node
        node.prev = self.head
        node.next = nxt
        nxt.prev = node

    def get(self, key: int) -> int:
        if key in self.cache:
            node = self.cache[key]
            self._remove(node)
            self._add_to_front(node)
            return node.val
        return -1

    def put(self, key: int, value: int) -> None:
        if key in self.cache:
            node = self.cache[key]
            node.val = value
            self._remove(node)
            self._add_to_front(node)
        else:
            node = _Node(key, value)
            self.cache[key] = node
            self._add_to_front(node)
            if len(self.cache) > self.capacity:
                # evict LRU (least recently used) – node just before tail
                lru = self.tail.prev
                self._remove(lru)
                del self.cache[lru.key]


class MinStack:
    """Stack supporting push, pop, top, and get_min in O(1)."""
    def __init__(self):
        self.stack = []
        self.min_stack = []

    def push(self, val: int) -> None:
        self.stack.append(val)
        if not self.min_stack:
            self.min_stack.append(val)
        else:
            self.min_stack.append(min(val, self.min_stack[-1]))

    def pop(self) -> None or int:
        if self.stack:
            self.min_stack.pop()
            return self.stack.pop()

    def top(self) -> int:
        if self.stack:
            return self.stack[-1]

    def get_min(self) -> int:
        if self.min_stack:
            return self.min_stack[-1]


class Heap:
    """Binary heap supporting min or max behaviour via factor."""
    def __init__(self, is_min: bool = True):
        self.heap = []
        self.factor = 1 if is_min else -1

    def push(self, val: int) -> None:
        v = val * self.factor
        self.heap.append(v)
        self._sift_up(len(self.heap) - 1)

    def pop(self) -> int:
        if not self.heap:
            return None
        if len(self.heap) == 1:
            return self.heap.pop() * self.factor
        root = self.heap[0]
        self.heap[0] = self.heap.pop()
        self._sift_down(0)
        return root * self.factor

    def peek(self) -> int:
        if not self.heap:
            return None
        return self.heap[0] * self.factor

    def __len__(self) -> int:
        return len(self.heap)

    def _sift_up(self, idx: int) -> None:
        while idx > 0:
            parent = (idx - 1) // 2
            if self.heap[idx] < self.heap[parent]:
                self.heap[idx], self.heap[parent] = self.heap[parent], self.heap[idx]
                idx = parent
            else:
                break

    def _sift_down(self, idx: int) -> None:
        n = len(self.heap)
        while True:
            left = 2 * idx + 1
            right = 2 * idx + 2
            smallest = idx
            if left < n and self.heap[left] < self.heap[smallest]:
                smallest = left
            if right < n and self.heap[right] < self.heap[smallest]:
                smallest = right
            if smallest != idx:
                self.heap[idx], self.heap[smallest] = self.heap[smallest], self.heap[idx]
                idx = smallest
            else:
                break


def find_median(stream: list[int]) -> list[float]:
    """
    Running median after each element from *stream*.
    Returns a list of floats.
    """
    # max-heap for lower half, min-heap for upper half
    small = Heap(is_min=False)   # max-heap
    large = Heap(is_min=True)    # min-heap
    result = []

    for num in stream:
        if len(small) == 0 or num <= small.peek():
            small.push(num)
        else:
            large.push(num)

        # balance: small can have at most one extra element
        if len(small) > len(large) + 1:
            large.push(small.pop())
        elif len(large) > len(small):
            small.push(large.pop())

        # compute median
        if len(small) == len(large):
            median = (small.peek() + large.peek()) / 2.0
        else:
            median = float(small.peek())
        result.append(median)

    return result
