```python
from flask import Flask, request, jsonify
from typing import List, Dict, Optional
from datetime import datetime

app = Flask(__name__)

# In-memory database
books: List[Dict] = []

class Book:
    def __init__(self, id: int, title: str, author: str, isbn: str, price: float):
        self.id = id
        self.title = title
        self.author = author
        self.isbn = isbn
        self.price = price
        self.created_at = datetime.now().isoformat()

    def to_dict(self) -> Dict:
        return {
            "id": self.id,
            "title": self.title,
            "author": self.author,
            "isbn": self.isbn,
            "price": self.price,
            "created_at": self.created_at
        }

def validate_book_data(data: Dict) -> bool:
    required_fields = ['title', 'author', 'isbn', 'price']
    for field in required_fields:
        if field not in data:
            return False
    if not isinstance(data['price'], (int, float)):
        return False
    if data['price'] <= 0:
        return False
    return True

@app.route('/books', methods=['GET'])
def get_books():
    page = request.args.get('page', 1, type=int)
    per_page = request.args.get('per_page', 10, type=int)
    
    if page < 1:
        page = 1
    if per_page < 1:
        per_page = 10
    
    start = (page - 1) * per_page
    end = start + per_page
    
    books_list = books[start:end]
    
    return jsonify({
        "books": [b.to_dict() for b in books_list],
        "total": len(books_list),
        "page": page,
        "per_page": per_page
    })

@app.route('/books/<int:book_id>', methods=['GET'])
def get_book(book_id: int):
    book = next((b for b in books if b['id'] == book_id), None)
    if not book:
        return jsonify({"error": "Book not found"}), 404
    return jsonify(book.to_dict())

@app.route('/books', methods=['POST'])
def create_book():
    data = request.get_json()
    
    if not validate_book_data(data):
        return jsonify({"error": "Validation failed"}), 422
    
    new_book = Book(
        id=len(books)+1,
        title=data['title'],
        author=data['author'],
        isbn=data['isbn'],
        price=data['price']
    )
    
    books.append(new_book.to_dict())
    return jsonify(new_book.to_dict()), 201

@app.route('/books/<int:book_id>', methods=['PUT'])
def update_book(book_id: int):
    book = next((b for b in books if b['id'] == book_id), None)
    if not book:
        return jsonify({"error": "Book not found"}), 404
    
    data = request.get_json()
    
    if not validate_book_data(data):
        return jsonify({"error": "Validation failed"}), 422
    
    book['title'] = data.get('title', book['title'])
    book['author'] = data.get('author', book['author'])
    book['isbn'] = data.get('isbn', book['isbn'])
    book['price'] = data.get('price', book['price'])
    
    return jsonify(book.to_dict()), 200

@app.route('/books/<int:book_id>', methods=['DELETE'])
def delete_book(book_id: int):
    book = next((b for b in books if b['id'] == book_id), None)
    if not book:
        return jsonify({"error": "Book not found"}), 404
    
    books.remove(book)
    return jsonify({"message": "Book deleted"}), 200

if __name__ == '__main__':
    app.run(debug=True)
```
